TypeScript / Mongoose src/models/client.model.ts (example full model) 
import { Schema, model, Document, Types } from 'mongoose';

export interface IClient extends Document {
  trainerId: Types.ObjectId;
  name: string;
  email: string;
  phone?: string;
  status: 'invited' | 'active' | 'paused' | 'completed';
  goals: string[];
  fitnessLevel: 'beginner' | 'intermediate' | 'advanced';
  sessionCount: number;
  engagementScore: number;
  lastCheckInAt?: Date;
  isDeleted: boolean;
}

const ClientSchema = new Schema<IClient>(
  {
    trainerId:  { type: Schema.Types.ObjectId, ref: 'User', required: true, index: true },
    name:       { type: String, required: true, trim: true },
    email:      { type: String, required: true, lowercase: true, trim: true },
    phone:      { type: String },
    status:     { type: String, enum: ['invited','active','paused','completed'],
                  default: 'invited', index: true },
    goals:      [{ type: String }],
    fitnessLevel: { type: String, enum: ['beginner','intermediate','advanced'] },
    sessionCount:    { type: Number, default: 0 },
    engagementScore: { type: Number, default: 100, min: 0, max: 100 },
    lastCheckInAt:   { type: Date, index: true },
    inviteToken:     { type: String, index: true, sparse: true },
    userId:          { type: Schema.Types.ObjectId, ref: 'User', sparse: true },
    fcmTokens:       [{ type: String }],
    isDeleted:       { type: Boolean, default: false, index: true },
  },
  { timestamps: true }
);

// Compound indexes for common query patterns
ClientSchema.index({ trainerId: 1, status: 1 });
ClientSchema.index({ trainerId: 1, lastCheckInAt: 1 });
ClientSchema.index({ trainerId: 1, isDeleted: 1 });

export const Client = model<IClient>('Client', ClientSchema);

TypeScript src/middleware/auth.middleware.ts 
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';
import { User } from '../models/user.model';
import { AppError } from '../utils/api-error';

// Extends Express Request to include authed user
declare global {
  namespace Express {
    interface Request {
      user?: { id: string; plan: string; subscriptionStatus: string };
    }
  }
}

export const requireAuth = async (req: Request, res: Response, next: NextFunction) => {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return next(new AppError('No token', 401));

  try {
    const decoded = jwt.verify(token, process.env.JWT_ACCESS_SECRET!) as any;
    req.user = { id: decoded.sub, plan: decoded.plan, subscriptionStatus: decoded.status };
    next();
  } catch {
    next(new AppError('Token expired', 401));
  }
};

// Ensures PT has active/trialing subscription
export const requireActiveSubscription = (req: Request, res: Response, next: NextFunction) => {
  const allowed = ['active', 'trialing'];
  if (!allowed.includes(req.user?.subscriptionStatus ?? '')) {
    return next(new AppError('Subscription required', 402));
  }
  next();
};

TypeScript src/middleware/validate.ts (Zod validation) 
import { Request, Response, NextFunction } from 'express';
import { AnyZodObject, ZodError } from 'zod';

export const validate = (schema: AnyZodObject) =>
  async (req: Request, res: Response, next: NextFunction) => {
    try {
      await schema.parseAsync({
        body: req.body,
        query: req.query,
        params: req.params,
      });
      next();
    } catch (err) {
      if (err instanceof ZodError) {
        res.status(400).json({
          success: false,
          message: 'Validation failed',
          errors: err.errors.map(e => ({ field: e.path.join('.'), message: e.message })),
        });
      }
    }
  };

// Usage: router.post('/clients', validate(createClientSchema), createClient)

TypeScript src/queues/workers/checkin-sender.worker.ts 
import { checkinQueue } from '../queue';
import { Client } from '../../models/client.model';
import { CheckIn } from '../../models/check-in.model';
import { notificationService } from '../../services/notification.service';
import { startOfWeek } from 'date-fns';

checkinQueue.process('weekly-send', async (job) => {
  const { trainerId } = job.data;

  // Find all active clients for this trainer
  const clients = await Client.find({
    trainerId, status: 'active', isDeleted: false
  }).select('_id name fcmTokens');

  const weekOf = startOfWeek(new Date(), { weekStartsOn: 1 });

  for (const client of clients) {
    // Upsert to avoid duplicate check-ins
    await CheckIn.findOneAndUpdate(
      { trainerId, clientId: client._id, weekOf },
      { $setOnInsert: { status: 'pending', sentAt: new Date() } },
      { upsert: true, new: true }
    );

    // Send push notification to client
    if (client.fcmTokens?.length) {
      await notificationService.sendPush(client.fcmTokens, {
        title: 'Weekly Check-in 📊',
        body: `Time for your weekly check-in with your trainer!`,
        data: { screen: 'checkin', weekOf: weekOf.toISOString() },
      });
    }
  }
});

// This worker is spawned for each PT's timezone group
// Scheduled via Bull cron: '0 8 * * 1' (Every Monday 8am)
